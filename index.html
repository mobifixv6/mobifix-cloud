-- =============================================
-- Mobifix PRO — إعداد قاعدة البيانات السحابية
-- انسخ هذا الكود كاملاً والصقه في SQL Editor
-- =============================================

-- 1. إنشاء جدول المزامنة الرئيسي
CREATE TABLE IF NOT EXISTS sync_data (
  center_name  TEXT PRIMARY KEY,
  devices      JSONB NOT NULL DEFAULT '[]',
  employees    JSONB NOT NULL DEFAULT '[]',
  attendance   JSONB NOT NULL DEFAULT '[]',
  expenses     JSONB NOT NULL DEFAULT '[]',
  settings     JSONB NOT NULL DEFAULT '{}',
  last_updated BIGINT NOT NULL DEFAULT 0,
  metadata     JSONB DEFAULT '{}'
);

-- 2. تفعيل نظام الأمان (Row Level Security)
ALTER TABLE sync_data ENABLE ROW LEVEL SECURITY;

-- 3. السماح بالوصول العام للمنظومة
DROP POLICY IF EXISTS "mobifix_public_access" ON sync_data;
CREATE POLICY "mobifix_public_access"
  ON sync_data
  FOR ALL
  USING (true)
  WITH CHECK (true);

-- 4. إنشاء دالة التنشيط التلقائي (تمنع توقف السحابة)
CREATE EXTENSION IF NOT EXISTS pg_cron;

CREATE OR REPLACE FUNCTION mobifix_keepalive()
RETURNS void
LANGUAGE plpgsql AS $$
DECLARE
  v_last_sync    BIGINT;
  v_now_ms       BIGINT;
  v_five_days_ms BIGINT := 432000000;
BEGIN
  v_now_ms := (extract(epoch FROM now()) * 1000)::bigint;
  
  SELECT COALESCE(MAX(last_updated), 0)
  INTO v_last_sync
  FROM sync_data
  WHERE center_name != 'MOBIFIX_HEARTBEAT';

  -- لو مر 5 أيام بدون أي نشاط → نكتب سجل heartbeat
  IF (v_now_ms - v_last_sync) > v_five_days_ms THEN
    INSERT INTO sync_data (center_name, last_updated, metadata)
    VALUES (
      'MOBIFIX_HEARTBEAT',
      v_now_ms,
      jsonb_build_object(
        'source', 'auto_keepalive',
        'reason', '5_days_no_activity',
        'triggered_at', to_char(now(), 'YYYY-MM-DD HH24:MI:SS')
      )
    )
    ON CONFLICT (center_name) DO UPDATE SET
      last_updated = EXCLUDED.last_updated,
      metadata     = EXCLUDED.metadata;
  END IF;
END;
$$;

-- 5. جدولة التنشيط يومياً الساعة 12 ظهراً
SELECT cron.unschedule(jobid) FROM cron.job WHERE jobname = 'mobifix-keepalive';
SELECT cron.schedule(
  'mobifix-keepalive',
  '0 12 * * *',
  $$SELECT mobifix_keepalive()$$
);

-- ✅ اكتمل الإعداد بنجاح
▶️
تشغيل الكود
  • بعد لصق الكود، اضغط زر "Run" الأخضر في أسفل أو أعلى الصفحة
  • انتظر بضع ثوانٍ حتى تظهر النتيجة
  • لو ظهرت رسالة باللون الأخضر → نجح كل شيء ✅
  • لو ظهر خطأ يذكر pg_cron فقط → لا تقلق، الجدول الرئيسي نجح وهو الأهم
⚠️ لو ظهر خطأ باللون الأحمر لا يذكر pg_cron — أعد نسخ الكود ولصقه والتأكد إنه كامل بدون حذف أي سطر، ثم اضغط Run مرة أخرى

الحصول على مفاتيح الربط

رابط + مفتاح — كلاهما في نفس المكان

🔍
كيف تصل لصفحة المفاتيح؟

الطريقة الأسرع:

🔍 شريط البحث العلوي ← اكتب: API ← اختر: Project API keys

أو من القائمة الجانبية: Settings ⚙️ ← API

🔗
أولاً — رابط المشروع (Project URL)
  • ابحث عن قسم "Project URL" في الصفحة
  • الرابط شكله هكذا:
    https://abcdefghijkl.supabase.co
  • اضغط Copy بجانبه أو انسخه يدوياً كاملاً
⚠️ لا تأخذ الرابط من شريط عنوان المتصفح — خذه فقط من قسم "Project URL" في صفحة API
🔑
ثانياً — مفتاح الوصول (API Key)
  • في نفس الصفحة، ابحث عن قسم "Project API keys"
  • ستجد مفتاحين — نريد anon و public فقط
  • اضغط Copy بجانب المفتاح الذي يكتب عنده anon و public
  • المفتاح طويل جداً ويبدأ بـ eyJhbGci...
🚫 لا تأخذ مفتاح service_role — هذا مفتاح سري وخطير. خذ فقط anon / public
✅ الآن عندك: رابط المشروع + مفتاح anon — هذا كل ما تحتاجه للخطوة الأخيرة
