-- =============================================
-- Mobifix PRO — إعداد قاعدة البيانات السحابية
-- =============================================

CREATE TABLE IF NOT EXISTS sync_data (
  center_name  TEXT PRIMARY KEY,
  devices      JSONB NOT NULL DEFAULT '[]',
  employees    JSONB NOT NULL DEFAULT '[]',
  attendance   JSONB NOT NULL DEFAULT '[]',
  expenses     JSONB NOT NULL DEFAULT '[]',
  settings     JSONB NOT NULL DEFAULT '{}',
  last_updated BIGINT NOT NULL DEFAULT 0,
  metadata     JSONB DEFAULT '{}'
);

ALTER TABLE sync_data ENABLE ROW LEVEL SECURITY;

DROP POLICY IF EXISTS "mobifix_public_access" ON sync_data;
CREATE POLICY "mobifix_public_access"
  ON sync_data
  FOR ALL
  USING (true)
  WITH CHECK (true);

CREATE EXTENSION IF NOT EXISTS pg_cron;

CREATE OR REPLACE FUNCTION mobifix_keepalive()
RETURNS void
LANGUAGE plpgsql AS $$
DECLARE
  v_last_sync    BIGINT;
  v_now_ms       BIGINT;
  v_five_days_ms BIGINT := 432000000;
BEGIN
  v_now_ms := (extract(epoch FROM now()) * 1000)::bigint;
  
  SELECT COALESCE(MAX(last_updated), 0)
  INTO v_last_sync
  FROM sync_data
  WHERE center_name != 'MOBIFIX_HEARTBEAT';

  IF (v_now_ms - v_last_sync) > v_five_days_ms THEN
    INSERT INTO sync_data (center_name, last_updated, metadata)
    VALUES (
      'MOBIFIX_HEARTBEAT',
      v_now_ms,
      jsonb_build_object(
        'source', 'auto_keepalive',
        'reason', '5_days_no_activity',
        'triggered_at', to_char(now(), 'YYYY-MM-DD HH24:MI:SS')
      )
    )
    ON CONFLICT (center_name) DO UPDATE SET
      last_updated = EXCLUDED.last_updated,
      metadata     = EXCLUDED.metadata;
  END IF;
END;
$$;

SELECT cron.unschedule(jobid) FROM cron.job WHERE jobname = 'mobifix-keepalive';
SELECT cron.schedule(
  'mobifix-keepalive',
  '0 12 * * *',
  $$SELECT mobifix_keepalive()$$
);
